```c
#include <stdio.h>
#define SIZE 3

typedef struct {
    int a[SIZE]; // a[3]
    int front;
    int rear;
} Queue;

void enq(Queue* q, int val){
    q->a[q->rear] = val; // q의 rear에 val을 넣는다
    q->rear = (q->rear + 1) % SIZE; // rear를 rear+1 % 3
}

int deq(Queue* q) {
    int val = q->a[q->front]; // val = front
    q->front = (q->front + 1) % SIZE; // front를 바꾼다.
    return val;
}

int main() {
    Queue q = {{0}, 0, 0};
    enq(&q,1); // q={{1},0,1}
    enq(&q,2); // q={{1,2},0,2}
    deq(&q); // q=
    enq(&q, 3);
    int first = deq(&q);
    int second = deq(&q);
    printf("%d 그리고 %d", first, second);
    return 0;
}
```
