11. 최종정답 : 13
     **이거 보면 set()함수에서 들어가는 순서를 생각해야함 그리고 (i+1)이라서 달라짐..**
```c
#include <stdio.h>

#include <stdlib.h>

void set(int** arr, int* data, int rows=3, int cols=3) {
	// for (int i = 0; i < 9; ++i)
    for (int i = 0; i < rows * cols; ++i) {
        arr[((i + 1) / rows) % rows][(i + 1) % cols] = data[i];
        // arr[(0/3)%3][(0+1)%3] = data[0] -> arr[0][1] = data[0]
        // arr[(1/3)%3][(1+1)%3] = data[1] -> arr[0][2] = data[1]
        // arr[(2/3)%3][(2+1)%3] = data[2] -> arr[0][0] = data[3]
        // arr[(3/3)%3][(0+1)%3] = data[0] -> arr[0][1] = data[0]
        
        
        // -> ** 위에 보면 (i+1) 안한 것을 볼 수 있음!!!! **
    }
}

int main() {
    int rows = 3, cols = 3, sum = 0;
    int data[] = {5, 2, 7, 4, 1, 8, 3, 6, 9}; 
    
    int** arr;
    arr = (int**) malloc(sizeof(int*) * rows);
    
    for (int i = 0; i < cols; i++) {
        arr[i] = (int*) malloc(sizeof(int) * cols);
    }

    set(arr, data, rows, cols);

    for (int i = 0; i < rows * cols; i++) {
        sum += arr[i / rows][i % cols] * (i % 2 == 0 ? 1 : -1);
        // arr[0/3][0%3] * (i가 짝수면 1/-2) -> arr[0][0]
        // arr[1/3][1%3] * -1 -> arr[0][1] * -1
        // arr[2/3][2%3] * 1 -> arr[2][2]
        // arr[3/3][3%3] * -1 -> arr[1][0] * -1
        // arr[4/3][4%3] -> arr[1][1]
        // arr[5/3][5%3] -> arr[1][2]*-1
    }

    for(int i=0; i<rows; i++) {
        free(arr[i]);
    }

    free(arr);

    printf("%d", sum);
}
```

    Data* new_node = (Data*)malloc(sizeof(Data)); // 새로운 Data구조체 생성
    new_node->value = value; // value 값 할당

Data* reconnect(Data* head, int value) {
    if (head == NULL || head->value == value) return head;
    
    Data *prev = NULL, *curr = head;
    
    // curr->value가 3이면 멈추게 됨. 즉 value가 3인 Data가 curr가 됨.
    while (curr != NULL && curr->value != value) {
        prev = curr;
        curr = curr->next;
    }

    if (curr != NULL && prev != NULL) {
        prev->next = curr->next;
        curr->next = head;
        head = curr;
    }

    return head;
}
