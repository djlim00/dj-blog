```java title:"키순서"
import java.io.*;  
import java.util.StringTokenizer;  
  
public class 플로이드워샬_키순서 {  
  
    static BufferedReader br = new BufferedReader(new InputStreamReader(System.in));  
    static BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));  
    static StringTokenizer st;  
  
    static int N, M;  
    static boolean[][] connected;  
    static int answer = 0;  
  
    public static void main(String[] args) throws IOException {  
        st = new StringTokenizer(br.readLine());  
        N = Integer.parseInt(st.nextToken());  
        M = Integer.parseInt(st.nextToken());  
        connected = new boolean[N][N];  
  
        for (int i = 0; i < M; i++) {  
            st = new StringTokenizer(br.readLine());  
            int a = Integer.parseInt(st.nextToken()) - 1; // a가 더 작음  
            int b = Integer.parseInt(st.nextToken()) - 1; // b가 a보다 큼  
            connected[a][b] = true;  
        }  
  
        // 모든 노드와 연결이 되어있다면(내쪽에서 쏘거나, 반대에서 날 쏘면) -> 등수를 정확하게 알 수 있음.  
        // 플로이드 워샬로 모든 노드의 모든 다른 노드에 대한 연결 여부를 확인  
        for (int k = 0; k < N; k++) {  
            for (int i = 0; i < N; i++) {  
                for (int j = 0; j < N; j++) {  
                    if (connected[i][k] && connected[k][j]) {  
                        connected[i][j] = true;  
                    }  
                }  
            }  
        }  
  
        int[] conCnt = new int[N];  
        for (int i = 0; i < N; i++) {  
            for (int j = 0; j < N; j++) {  
                // 내가 쏘기 || 걔가 날 쏘기  
                if (connected[i][j] || connected[j][i]) {  
                    conCnt[i]++;  
                }  
            }  
        }  
  
        for (int i = 0; i < N; i++) {  
            // 전부 연결되어 있는 애  
            if (conCnt[i] == N - 1) {  
                answer++;  
            }  
        }  
  
        bw.write(String.valueOf(answer));  
        bw.flush();  
    }  
}
```
