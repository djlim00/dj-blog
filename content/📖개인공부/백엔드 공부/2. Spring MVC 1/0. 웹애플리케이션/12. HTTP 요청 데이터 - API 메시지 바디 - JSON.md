# "HelloData 클래스로 변환하기" - Jackson 라이브러리의 ObjectMapper

```java title:"HelloData 객체로 변환하기 - ObjectMapper, hl:5,14

@WebServlet(name="requestBodyJsonServlet",urlPatterns = "/request-body-json")  
public class RequestBodyJsonServlet extends HttpServlet {  
  
    private ObjectMapper objectMappers= new ObjectMapper();  
  
    @Override  
    protected void service(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {  
        ServletInputStream inputStream = request.getInputStream();  
        String messageBody = StreamUtils.copyToString(inputStream, StandardCharsets.UTF_8);  
  
        System.out.println("messageBody = " + messageBody);  
  
        HelloData helloData = objectMappers.readValue(messageBody, HelloData.class);  
  
        System.out.println("helloData.username = " + helloData.getUsername());  
        System.out.println("helloData.age = " + helloData.getAge());  
  
        response.getWriter().write("OK");  
  
  
    }  
}
```
