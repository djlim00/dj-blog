```java title:"ResponseHeaderServlet->응답코드 설정 예시", hl:36-37
package hello.servlet.basic.response;  
  
import jakarta.servlet.ServletException;  
import jakarta.servlet.annotation.WebServlet;  
import jakarta.servlet.http.Cookie;  
import jakarta.servlet.http.HttpServlet;  
import jakarta.servlet.http.HttpServletRequest;  
import jakarta.servlet.http.HttpServletResponse;  
  
import java.io.IOException;  
import java.io.PrintWriter;  
  
@WebServlet(name="responseHeaderServlet",urlPatterns = "/response-header")  
public class ResponseHeaderServlet extends HttpServlet {  
    @Override  
    protected void service(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {  
        // [status-line]  
        response.setStatus(HttpServletResponse.SC_OK); // 그냥 200으로 쓰는 것 보다 이게 좋음  
  
  
        //[response-header]  
        response.setHeader("Content-Type","text/plain;charset=utf-8");  
        response.setHeader("Cache-Control","no-cache, no-store, must-revalidate");  
        response.setHeader("Pragma","no-cache");  
        response.setHeader("my-header","hello");  
  
        // [헤더 편의 메서드]  
        content(response);  
  
        // [쿠키 편의 메서드]  
        cookie(response);  
  
        // [리다이렉트 편의 메서드]  
        redirect(response);  
  
        PrintWriter printWriter = response.getWriter();  
        printWriter.write("OK");  
    }  
  
    private void content(HttpServletResponse response) {  
        //Content-Type: text/plain;charset=utf-8  
        //Content-Length: 2        //response.setHeader("Content-Type", "text/plain;charset=utf-8");        response.setContentType("text/plain");  
        response.setCharacterEncoding("utf-8");  
        //response.setContentLength(2); //(생략시 자동 생성)  
    }  
  
    private void cookie(HttpServletResponse response) {  
        //Set-Cookie: myCookie=good; Max-Age=600;  
        //response.setHeader("Set-Cookie", "myCookie=good; Max-Age=600");        Cookie cookie = new Cookie("myCookie", "good");  
        cookie.setMaxAge(600); //600초  
        response.addCookie(cookie);  
    }  
  
  
    private void redirect(HttpServletResponse response) throws IOException {  
        //Status Code 302  
        //Location: /basic/hello-form.html        //response.setStatus(HttpServletResponse.SC_FOUND); //302        //response.setHeader("Location", "/basic/hello-form.html");        response.sendRedirect("/basic/hello-form.html");}  
}
```

- response.setHeader()를 이용해서 가능하다.
- 근데<font color="#ffff00"> setHeader()를 하면 너무 귀찮으니까, 각각의 헤더에 대한 함수를 사용하면 편하다.</font>
	- ex) setCharactoerEncoding(), setContentType(), setMaxAge(), addCookiee()...etc
- 리다이렉트는 <font color="#ffff00">response.sendRedirect("/basic/hello-form.html");</font>를 사용하면 편하게 쓸 수 있다.

- <font color="#ffff00"> PrintWriter객체를 이용해서 메세지 바디</font>를 편리하게 만들 수 있다
