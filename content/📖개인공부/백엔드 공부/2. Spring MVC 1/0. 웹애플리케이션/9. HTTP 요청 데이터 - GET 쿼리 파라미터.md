```java title:"쿼리 파라미터 조회 방식", hl:20,27,35
package hello.servlet.basic.request;  
  
import jakarta.servlet.ServletException;  
import jakarta.servlet.annotation.WebServlet;  
import jakarta.servlet.http.HttpServlet;  
import jakarta.servlet.http.HttpServletRequest;  
import jakarta.servlet.http.HttpServletResponse;  
  
import java.io.IOException;  
import java.util.Enumeration;  
  
/**  
 * 1. 파라미터 전송 기능  
 * http://localhost:8080/request-param?username=hello&age=20  
 */  
@WebServlet(name="requestParamSerlet", urlPatterns = "/request-param")  
public class RequestParamServlet extends HttpServlet {  
    @Override  
    protected void service(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {  
        System.out.println("[전체 파라미터 조회] - start");  
  
        request.getParameterNames().asIterator()  
                        .forEachRemaining(paramName -> System.out.println(paramName + "="+request.getParameter(paramName)));  
  
        System.out.println("[전체 파라미터 조회] - end\n");  
  
        System.out.println("[단일 파마미터 조회] - start");  
        String username = request.getParameter("username");  
        String age = request.getParameter("age");  
  
        System.out.println("username = " + username);  
        System.out.println("age = " + age);  
        System.out.println("[단일 파마미터 조회] - end\n");  
  
        System.out.println("[이름이 같은 복수 파라미터 조회] - start");  
        String[] usernames = request.getParameterValues("username");  
        for(String name:usernames){  
            System.out.println("username = " + name);  
        }  
        System.out.println("[이름이 같은 복수 파라미터 조회] - end");  
  
        response.getWriter().write("OK");  
  
  
    }  
}
```

1. 파라미터 이름을 출력하는 방식
2. 단일 파라미터의 값을 출력
3. 같은 파라미터의 값을 배열로 받기

```java title:"결과"

[단일 파마미터 조회] - end

```

#### 복수 파라미터에서 단일 파라미터 조회

- "username=hello&username=hello2" 이렇게 두개나 나오면 어떻게 되나..?
	- req.getParameterValues()를 사용

- <font color="#ffff00">근데 중복으로 보내는 경우는 거의 없음</font>
- req.getParameter()를 잘 기억하자
