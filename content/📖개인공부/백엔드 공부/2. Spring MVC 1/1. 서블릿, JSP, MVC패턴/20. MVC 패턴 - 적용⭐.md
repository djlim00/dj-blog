```java title:"MVC패턴 - 회원가입 폼", hl:18
package hello.servlet.web.servletmvc;  
  
import jakarta.servlet.RequestDispatcher;  
import jakarta.servlet.ServletException;  
import jakarta.servlet.annotation.WebServlet;  
import jakarta.servlet.http.HttpServlet;  
import jakarta.servlet.http.HttpServletRequest;  
import jakarta.servlet.http.HttpServletResponse;  
  
import java.io.IOException;  
  
@WebServlet(name="mvcMemberFormServlet",urlPatterns = "/servlet-mvc/members/new-form")  
public class MvcMemberFormServlet extends HttpServlet {  
    @Override  
    protected void service(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {  
        String viewPath = "/WEB-INF/views/new-form.jsp";  
        RequestDispatcher requestDispatcher = request.getRequestDispatcher(viewPath);  
        requestDispatcher.forward(request,response);  
    }  
}
```

- JSP파일의 경로를 이용한다
- RequestDispatcher를 이용해서 request.getRequestDispatcher를 이용해서 해당 경로로 이동하게 한다,
- forward()를 이용해서 서버 내부 통신을 통해서 다른 페이지(여기서는 save)로 이동하게 한다
	- 즉 (클라이언트 -> 서버 -> 클라이언트 -> 서버)가 아니라 (클라이언트 -> 서버 -> 서버)로 바로 서버 안에서 찾아서 바로 반환한다.
	- 클라이언트는 이런 과정을 전혀 알 수 없다(서버 내부에서만 일어나는 일)

```java title:"JSP를 이용한 회원 목록 조회", hl:37,50-57
package hello.servlet.web.servletmvc;  
  
import hello.servlet.domain.member.Member;  
import hello.servlet.domain.member.MemberRepository;  
import jakarta.servlet.RequestDispatcher;  
import jakarta.servlet.ServletException;  
import jakarta.servlet.annotation.WebServlet;  
import jakarta.servlet.http.HttpServlet;  
import jakarta.servlet.http.HttpServletRequest;  
import jakarta.servlet.http.HttpServletResponse;  
  
import java.io.IOException;  
import java.util.List;  
  
@WebServlet(name="mvcMemberListServlet", urlPatterns = "/servlet-mvc/members")  
public class MvcMemberListServlet extends HttpServlet {  
  
    MemberRepository memberRepository = MemberRepository.getInstance();  
  
    @Override  
    protected void service(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {  
  
        List<Member> members = memberRepository.finfAll();  
  
        request.setAttribute("members",members);  
  
        String viewPath = "/WEB-INF/views/members.jsp";  
        RequestDispatcher requestDispatcher = request.getRequestDispatcher(viewPath);  
        requestDispatcher.forward(request,response);  
    }  
}
```
