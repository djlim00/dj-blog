![[Pasted image 20250310133303.png]]
- <mark style="background: #FFF3A3A6;">위의 두 방식은 둘다 형식이 같기 때문에 구분없이 조회할 수 있다.</mark>
	- 간단히 "요청 파라미터 조회"라고 한다.

```java title:"Get, Post-html form 방식"
package hello.springmvc.basic.request;  
  
import jakarta.servlet.http.HttpServletRequest;  
import jakarta.servlet.http.HttpServletResponse;  
import lombok.extern.slf4j.Slf4j;  
import org.springframework.stereotype.Controller;  
import org.springframework.web.bind.annotation.RequestMapping;  
  
import java.io.IOException;  
  
@Slf4j  
@Controller  
public class RequestParamController {  
  
    @RequestMapping("/request-param-v1")  
    public void requestParamV1(HttpServletRequest req, HttpServletResponse res) throws IOException {  
        String username = req.getParameter("username");  
        int age = Integer.parseInt(req.getParameter("age"));  
        log.info("username={}, age={}",username, age);  
  
        // body에 값을 쓰기  
        res.getWriter().write("ok");  
    }  
}

/============================================================

<!DOCTYPE html>  
<html>  
<head>  
    <meta charset="UTF-8">  
    <title>Title</title>  
</head>  
<body>  
<form action="/request-param-v1" method="post">  
    username: <input type="text" name="username" />  
    age: <input type="text" name="age" />  
    <button type="submit">전송</button>  
</form>  
</body>  
</html>
```
- resources의 static에 있는 파일들은 그냥  열려있기 때문에 거기에 만들면 서버 띄우면 들어갈 수 있음
